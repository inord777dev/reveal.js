"Collection" Methods

_.forEach (_.each)

              
              const arr = [1, 2, 3, 4, 5]

              _.forEach(arr, x => x == 3 ? false : console.log(x))
              // => 1
              // => 2
              // => 3
              // => [1, 2, 3, 4, 5]

              arr.forEach(x => x == 3 ? false : console.log(x))
              // => 1
              // => 2
              // => 3
              // => 4
              // => 5
              // => undefined

_.map

              
              const arr = [{name: 'Andrei'}, {name: 'Vlad'}, {name: 'Kirill'}]

              _.map(arr, 'name')
              // => ['Andrei', 'Vlad', 'Kirill']

              arr.map(x => x.name)
              // => ['Andrei', 'Vlad', 'Kirill']

_.sample

              
                _.sample([1, 2, 3, 4]);
                // => 2

_.groupBy

              
                _.groupBy([6.1, 4.2, 6.3], Math.floor);
                // => { '4': [4.2], '6': [6.1, 6.3] }
                 
                // The `_.property` iteratee shorthand.
                _.groupBy(['one', 'two', 'three'], 'length');
                // => { '3': ['one', 'two'], '5': ['three'] }

_.keyBy

              
              var array = [ 
                { 'dir': 'left', 'code': 97 }, { 'dir': 'right', 'code': 100 } 
              ];
              _.keyBy(array, function(o) {
                return String.fromCharCode(o.code);
              });
              // => { 
                'a': { 'dir': 'left', 'code': 97 },
                'd': { 'dir': 'right', 'code': 100 }
              }
              _.keyBy(array, 'dir');
              // => { 
                'left': { 'dir': 'left', 'code': 97 },
                'right': { 'dir': 'right', 'code': 100 }
              }
"Array" Methods

_.pull

              
                var array = ['a', 'b', 'c', 'a', 'b', 'c'];
 
                _.pull(array, 'a', 'c');
                console.log(array);
                // => ['b', 'b']
  • _.without - does't mutate array.
  • _.remove - remove elements from an array by predicate.

_.uniq

              
                _.uniq([2, 1, 2]);
                // => [2, 1]

_.union

              
                _.union([2], [1, 2]);
                // => [2, 1]
"Function" Methods

_.after

              
                var saves = ['profile', 'settings'];
 
                var done = _.after(saves.length, function() {
                  console.log('done saving!');
                });
                 
                _.forEach(saves, function(type) {
                  asyncSave({ 'type': type, 'complete': done });
                });
                // => Logs 'done saving!' after the two async saves have completed.

_.before - opposite method

_.curry

              
                var abc = function(a, b, c) {
                  return [a, b, c];
                };
                var curried = _.curry(abc);
                curried(1)(2)(3);
                // => [1, 2, 3]
                curried(1, 2)(3);
                // => [1, 2, 3]
                curried(1, 2, 3);
                // => [1, 2, 3]
                // Curried with placeholders.
                curried(1)(_, 3)(2);
                // => [1, 2, 3]

_.debounce

              
                // Avoid costly calculations while the window size is in flux.
                jQuery(window).on('resize', _.debounce(calculateLayout, 150));
                 
                // Invoke `sendMail` when clicked, debouncing subsequent calls.
                jQuery(element).on('click', _.debounce(sendMail, 300, {
                  'leading': true,
                  'trailing': false
                }));

Arguments

  1. The function to throttle.
  2. The number of milliseconds to throttle invocations.
  3. The options object with keys specify invoking on the leading edge and the trailing edge of the timeout.

_.throttle

              
                // Avoid excessively updating the position while scrolling.
                jQuery(window).on('scroll', _.throttle(updatePosition, 100));
                // Invoke `renewToken` when the click event is fired,
                // but not more than once every 5 minutes.
                var throttled = _.throttle(renewToken, 300000, { 'trailing': false });
                jQuery(element).on('click', throttled);
                // Cancel the trailing throttled invocation.
                jQuery(window).on('popstate', throttled.cancel);

Arguments

  1. The function to throttle.
  2. The number of milliseconds to throttle invocations.
  3. The options object with keys specify invoking on the leading edge and the trailing edge of the timeout.

"Lang" Methods

_.cloneDeepWith

            
              function customizer(value) {
                if (_.isElement(value)) {
                  return value.cloneNode(true);
                }
              }
               
              var el = _.cloneDeepWith(document.body, customizer);
               
              console.log(el === document.body);
              // => false
              console.log(el.nodeName);
              // => 'BODY'
              console.log(el.childNodes.length);
              // => 20
