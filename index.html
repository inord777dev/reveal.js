Collection Methods

_.forEach (_.each)

              
              const arr = [1, 2, 3, 4, 5]

              _.forEach(arr, x => x == 3 ? false : console.log(x))
              => 1
              => 2
              => 3
              => [1, 2, 3, 4, 5]

              arr.forEach(x => x == 3 ? false : console.log(x))
              => 1
              => 2
              => 3
              => 4
              => 5
              => undefined

_.map

              
              const arr = [{name: 'Andrei'}, {name: 'Vlad'}, {name: 'Kirill'}]

              _.map(arr, 'name')
              => ['Andrei', 'Vlad', 'Kirill']

              arr.map(x => x.name)
              => ['Andrei', 'Vlad', 'Kirill']

_.groupBy

              
                _.groupBy([6.1, 4.2, 6.3], Math.floor);
                // => { '4': [4.2], '6': [6.1, 6.3] }
                 
                // The `_.property` iteratee shorthand.
                _.groupBy(['one', 'two', 'three'], 'length');
                // => { '3': ['one', 'two'], '5': ['three'] }
